export function ImageOne({ queryData }) {

    return <Image loading="lazy" decoding="async"
      src={placeholderImg}
      alt="Placeholder Image"
      class="w-full"
    />;
  }

const image = {
src: queryData.mediaItem1.nodes[0].mediaItemUrl ?? onImageError("No image found for ImageOne").src,
alt: queryData.mediaItem1.nodes[0].altText ?? "Placeholder Image",

width: queryData.mediaItem1.nodes[0].mediaDetails?.width ?? 500,
};

return (

<Image
  loading="lazy"
  decoding="async"

  width={image.width}

  alt={image.alt}
  class="w-full"
/>
); }
